    salvarBtn.addEventListener('click', () => {
      const texto = resumoTxt.value.trim();
      if(texto.length === 0) {
        alert('Escreva algo antes de salvar!');
        return;
      }
      localStorage.setItem('resumoDia', texto);
      msgSalvar.style.display = 'block';
      setTimeout(() => {
        msgSalvar.style.display = 'none';
      }, 2000);
    });
